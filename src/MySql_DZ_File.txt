create database if not exists boooks;

use boooks;

create table if not exists author(id int primary key auto_increment, name varchar(20) not null,surname varchar(20) not null);
create table if not exists book(book_id int primary key auto_increment,
title varchar(20) not null,pages int not null,
author_id int not null,
foreign key(author_id) references author (id));
